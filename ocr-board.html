  2. 위 <style>의 .answer-pad / .scratch-pad-wrap / .scratch-pad / .ans / .pads-row 등 CSS 복사
  3. 아래 <script id="ocr-board-lib"> 라이브러리 블록 전체 복사
  4. 아래 적용 패턴(makeProblems 스타일) 코드를 자기 페이지에서 호출

🎯 핵심 동작 (다른 OCR 보드와의 차이)

  1. 그리기 종료 즉시 자동 인식 — 350ms 디바운스 후 recognizeDigitsFromCanvas 호출, 결과를 정답 입력칸에 채움
  2. 결과가 빈 문자열이면 입력칸 보존 — if(rec) 가드로 학생이 직접 친 값이 지워지지 않음
  3. 정답 확인 클릭 시 OCR 강제 재실행 — 보드에 남아 있는 마지막 글자까지 반영. 이때 버튼 라벨은 잠깐 "인식 중…" 으로 변함
  4. race 방지 토큰 — 빠르게 다시 그리면 이전 OCR 결과는 무시
  5. 지우기 시 토큰 무효화 + 입력칸 비우기 — 비대기 중 OCR이 빈 값으로 덮어쓰지 않도록
// 적용 패턴 (각 입력칸 ↔ OCR 패드 페어 N개)
const ansInputs = [...document.querySelectorAll('input.ans')];   // 정답 입력칸 N개
const answerPads = [...document.querySelectorAll('canvas.answer-pad')];  // OCR 패드 N개
const ocrTokens = answerPads.map(() => ({ n: 0 }));   // race 방지 토큰

const padHandlers = answerPads.map((c, idx) => setupAnswerPad(c, {
  onChange: async () => {
    const tk = ocrTokens[idx];
    const myN = ++tk.n;
    const rec = await recognizeDigitsFromCanvas(c);
    if(myN !== tk.n) return;             // 더 새로운 그리기가 시작됨 — 무시
    if(ansInputs[idx] && rec){           // 빈 결과로 학생 입력 덮어쓰지 않음
      ansInputs[idx].value = rec;
      ansInputs[idx].classList.remove('ok','bad');
    }
  }
}));

// 지우기 버튼
document.querySelector('.clear-answer-pads').addEventListener('click', () => {
  padHandlers.forEach(h => h && h.clear());
  ocrTokens.forEach(tk => tk.n++);       // 대기 중 OCR 무효화
  answerPads.forEach(p => p.classList.remove('correct','wrong'));
  ansInputs.forEach(inp => { if(inp){ inp.value = ''; inp.classList.remove('ok','bad'); } });
});

// 정답 확인 — OCR을 한 번 더 강제 실행해서 최신 글자까지 반영한 뒤 검증
document.querySelector('#check').addEventListener('click', async () => {
  const btn = document.querySelector('#check');
  if(btn.disabled) return;
  btn.disabled = true;
  const orig = btn.textContent;
  btn.textContent = '인식 중...';
  await Promise.all(answerPads.map(async (pad, j) => {
    try {
      const rec = await recognizeDigitsFromCanvas(pad);
      if(rec && ansInputs[j]) ansInputs[j].value = rec;
    } catch(e){ /* 실패해도 입력값으로 검증 */ }
  }));
  btn.textContent = orig;
  btn.disabled = false;

  // 여기서 ansInputs[j].value 를 정답과 비교해 채점
});
